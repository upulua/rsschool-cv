function toCamelCase(str){
  let ans = "";
  let prev = false;
  for (let i = 0; i < str.length; i++) {
    if ((str[i] == '-') || (str[i] == '_')) {
      prev = true;
    } else {
      ans = prev ? ans + str[i].toUpperCase() : ans + str[i]; 
      prev = false;
    }
  }
  return ans;
}
